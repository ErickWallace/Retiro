⚙️ Guia de Configuração

Passo 1 — Criar a Planilha Google Sheets

  1. Acesse sheets.google.com e crie uma planilha nova.
  2. Renomeie a aba para Tarefas.
  3. O Apps Script criará o cabeçalho automaticamente na primeira execução.

Passo 2 — Criar o Apps Script

  1. Na planilha, clique em Extensões → Apps Script.
  2. Apague todo o código existente.
  3. Copie o código abaixo e cole no editor.
  4. Salve (Ctrl+S) e dê um nome ao projeto.
// ════════════════════════════════════════════════════
//  KANBAN DO EVENTO — Google Apps Script
//  Tudo via GET (evita CORS preflight)
// ════════════════════════════════════════════════════

const SHEET_NAME = "Tarefas";
const COLUNAS = ["id","categoria","titulo","descricao","responsaveis","data","status","criadoEm","atualizadoEm"];

function doGet(e) {
  try {
    const acao = e.parameter.acao || "listar";
    let resultado;

    if (acao === "listar") {
      resultado = listarTarefas();
    } else if (acao === "criar" || acao === "editar") {
      const tarefa = JSON.parse(decodeURIComponent(e.parameter.payload));
      resultado = acao === "criar" ? criarTarefa(tarefa) : editarTarefa(tarefa);
    } else if (acao === "excluir") {
      resultado = excluirTarefa(e.parameter.id);
    } else {
      resultado = { error: "Ação desconhecida: " + acao };
    }

    return ContentService
      .createTextOutput(JSON.stringify(resultado))
      .setMimeType(ContentService.MimeType.JSON);

  } catch (err) {
    return ContentService
      .createTextOutput(JSON.stringify({ error: err.message }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function listarTarefas() {
  const sheet = getSheet();
  const dados = sheet.getDataRange().getValues();
  if (dados.length <= 1) return { tarefas: [] };
  const cabecalho = dados[0];
  const tarefas = dados.slice(1).map(row => {
    const obj = {};
    cabecalho.forEach((col, i) => {
      let val = row[i];
      if (val instanceof Date) {
        if (col === "data") {
          const y = val.getFullYear();
          const m = String(val.getMonth() + 1).padStart(2, "0");
          const d = String(val.getDate()).padStart(2, "0");
          val = y + "-" + m + "-" + d;
        } else {
          val = val.toISOString();
        }
      }
      obj[col] = val || "";
    });
    return obj;
  });
  return { tarefas };
}

function criarTarefa(t) {
  const sheet = getSheet();
  const agora = new Date().toISOString();
  const linha = COLUNAS.map(col => {
    if (col === "criadoEm" || col === "atualizadoEm") return agora;
    return t[col] !== undefined ? t[col] : "";
  });
  sheet.appendRow(linha);
  return { ok: true, id: t.id };
}

function editarTarefa(t) {
  const sheet = getSheet();
  const dados = sheet.getDataRange().getValues();
  const cabecalho = dados[0];
  const idCol = cabecalho.indexOf("id");
  for (let i = 1; i < dados.length; i++) {
    if (String(dados[i][idCol]) === String(t.id)) {
      COLUNAS.forEach((col, j) => {
        if (col === "criadoEm") return;
        const val = col === "atualizadoEm" ? new Date().toISOString() : (t[col] !== undefined ? t[col] : dados[i][j]);
        sheet.getRange(i + 1, j + 1).setValue(val);
      });
      return { ok: true };
    }
  }
  return { error: "Tarefa não encontrada: " + t.id };
}

function excluirTarefa(id) {
  const sheet = getSheet();
  const dados = sheet.getDataRange().getValues();
  const cabecalho = dados[0];
  const idCol = cabecalho.indexOf("id");
  for (let i = 1; i < dados.length; i++) {
    if (String(dados[i][idCol]) === String(id)) {
      sheet.deleteRow(i + 1);
      return { ok: true };
    }
  }
  return { error: "Tarefa não encontrada: " + id };
}

function getSheet() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName(SHEET_NAME);
  if (!sheet) {
    sheet = ss.insertSheet(SHEET_NAME);
    sheet.appendRow(COLUNAS);
    const r = sheet.getRange(1, 1, 1, COLUNAS.length);
    r.setBackground("#1b4332");
    r.setFontColor("#ffffff");
    r.setFontWeight("bold");
    sheet.setFrozenRows(1);
  }
  return sheet;
}

Passo 3 — Publicar como Web App

  1. No editor do Apps Script, clique em Implantar → Nova implantação.
  2. Clique no ícone ⚙️ ao lado de "Tipo" e escolha App da Web.
  3. Configure:
    — Executar como: Eu mesmo
    — Quem tem acesso: Qualquer pessoa
  4. Clique em Implantar e autorize o acesso.
  5. Copie a URL do Web App gerada (começa com https://script.google.com/macros/s/…/exec).
⚠️ Atenção: Sempre que editar o código do Apps Script, crie uma nova implantação. Gerenciar a implantação existente não atualiza o código em produção.

Passo 4 — Colocar a URL neste arquivo

  1. Abra o index.html em um editor de texto.
  2. Localize esta linha no início do arquivo:
const APPS_SCRIPT_URL = "https://script.google.com/macros/s/AKfycbzrriq1x17UzqN0hRvVFcdBB4CTYfKSagnJ4DdD3KcRI6mYjITaZ1Zp68I27CxLJy6L/exec";
  1. Substitua COLE_A_URL_AQUI pela URL copiada no Passo 3.
  2. Salve o arquivo e faça upload no GitHub.

Passo 5 — Publicar no GitHub Pages

  1. Crie um repositório público no GitHub.
  2. Faça upload do index.html.
  3. Vá em Settings → Pages, selecione a branch main e salve.
  4. Aguarde alguns minutos e acesse: https://seu-usuario.github.io/nome-do-repo/
